504 /****************** BFS ********************/
505 /*
506  * Performs a breadth-first search (bfs) in a graph (this), starting
507  * from the vertex with the given source contents (source).
508  * Returns a vector with the contents of the vertices by bfs order.
509  */
510 template <class T>
511 std::vector<T> GraphExample<T>::bfs(const T & source) const {
512  std::vector<int> res;
513  // Get the source vertex
514  auto s = findVertex(source);
515  if (s == nullptr) {
516  return res;
517  }
518 
519  // Set that no vertex has been _visited yet
520  for (auto v : _vertexSet) {
521  v->setVisited(false);
522  }
523 
524  // Perform the actual BFS using a queue
525  std::queue<Vertex<T> *> q;
526  q.push(s);
527  s->setVisited(true);
528  while (!q.empty()) {
529  auto v = q.front();
530  q.pop();
531  res.push_back(v->getInfo());
532  for (auto & e : v->getAdj()) {
533  auto w = e->getDest();
534  if ( ! w->isVisited()) {
535  q.push(w);
536  w->setVisited(true);
537  }
538  }
539  }
540  return res;
541 }
